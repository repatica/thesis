\newcounter{comment}
\newcommand{\comment}[2]{}
\newcommand{\commentmarker}[1]{%
  \stepcounter{comment}%
  \llap{\raisebox{0.5\baselineskip}[0pt][0pt]{\makecommentmarker{#1}}}}
\newcommand{\makecommentmarker}[1]{%
  \begingroup
  \setlength{\fboxsep}{1pt}%
  \csname #1commentcolor\endcsname
  \fbox{\commentmarkerfont\thecomment}%
  \endgroup}
\newcommand{\commenttext}[2]{%
  \marginpar{%
    \raggedright
    \leavevmode
    \csname #1commentcolor\endcsname
    \makecommentmarker{#1}
    \begingroup\setlength{\fboxsep}{1pt}\fbox{\commentmarkerfont #1}\endgroup\
    \commenttextfont #2}}
\newcommand{\withcomments}{%
  \addtolength{\paperwidth}{1in}
  \addtolength{\marginparwidth}{1in}
  \renewcommand{\comment}[2]{\commentmarker{##1}\commenttext{##1}{##2}}}
